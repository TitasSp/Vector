427 iterator insert(const_iterator pos, size_type count, const value_type& value) {
428 if (count == 0) return const_cast<iterator>(pos);
429
430 size_type position = pos - begin();
431 if (position > sz) throw std::out_of_range("Vector::insert position out of range");
432
433 if (sz + count > cap) {
434 size_type new_cap = std::max(cap == 0 ? 1 : cap * 2, sz + count);
435
436 // Allocate new memory
437 pointer new_elem = allocate(new_cap);
438
439 try {
440 // Move construct elements before insertion point
441 for (size_type i = 0; i < position; ++i) {
442 construct(new_elem + i, std::move(elem[i]));
443 }
444
445 // Construct new elements
446 for (size_type i = 0; i < count; ++i) {
447 construct(new_elem + position + i, value);
448 }
449
450 // Move construct elements after insertion point
451 for (size_type i = position; i < sz; ++i) {
452 construct(new_elem + i + count, std::move(elem[i]));
453 }
454 } catch (...) {
455 // Clean up on exception
456 destroy_range(new_elem, new_elem + position + count + (sz - position));
457 deallocate(new_elem, new_cap);
458 throw;

461 // Clean up old storage
462 destroy_range(elem, elem + sz);
463 deallocate(elem, cap);
464
465 elem = new_elem;
466 cap = new_cap;
467 } else {
468 // Sufficient capacity, shift elements in place
469

471 for (size_type i = sz; i < sz + count; ++i) {
472 construct(elem + i, value);
473 }
